\section{Greedy Heuristic Pseudocode}

\Needspace{12cm}
\begin{algorithm}[H]
\caption{Greedy solver for regime adjustment and crew routing}
\label{alg:greedy_solver}
\begin{algorithmic}[1]
\Require Instance $I=(W,B,D)$, configuration $cfg=(\tau, Q_{\max}, C_{\max}^{g}, L_{\max}^{g}, m, crews)$
\Ensure Feasible solution $S$ (selected wells, new regimes, routes) or \textsc{Infeasible}

\State Initialize $S$ with empty selected wells, empty routes, and infeasible flag
\State Build map $W_b \gets \{w \in W : w.battery = b\}$ for each battery $b \in B$
\State $selCount \gets 0$, $selCost \gets 0$, $selLoss \gets 0$

\ForAll{$b \in B$}
    \State $maxProd \gets \sum_{w \in W_b} \frac{w.G}{w.r} \cdot 100$
    \State $curProd \gets \sum_{w \in W_b} w.G$
    \State $target \gets \min(b.Gpt, maxProd)$

    \If{$curProd < target(1-\tau)$}
        \State $needIncrease \gets$ true
        \State $gap \gets target - curProd$
    \ElsIf{$curProd > target(1+\tau)$}
        \State $needIncrease \gets$ false
        \State $gap \gets curProd - target$
    \Else
        \State \textbf{continue} 
    \EndIf

    \State Sort wells in $W_b$ according to method $m$:
    \Statex \hspace{1em} \texttt{priority\_cost}: order by regime ratio (ascending if increase, descending if decrease)
    \Statex \hspace{1em} \texttt{loss}: order by $(G-N)$ (ascending if increase, descending if decrease)
    \Statex \hspace{1em} \texttt{route}: nearest-neighbor route order from distance matrix $D$

    \State $batCost \gets 0$, $batLoss \gets 0$, $selected_b \gets \emptyset$

    \ForAll{$w$ in sorted $W_b$}
        \If{$gap \le 0$}
            \State \textbf{break}
        \EndIf

        \If{$needIncrease$}
            \State $room \gets \frac{w.G}{w.r} \cdot 100 - w.G$
            \If{$room \le 0$}
                \State \textbf{continue}
            \EndIf
            \If{$gap \ge room$}
                \State $r' \gets 100$, $\Delta \gets room$
            \Else
                \State $r' \gets w.r + \frac{gap}{(w.G/w.r)\cdot 100}$, $\Delta \gets gap$
            \EndIf
        \Else
            \State $room \gets w.G$
            \If{$room \le 0$}
                \State \textbf{continue}
            \EndIf
            \If{$gap \ge room$}
                \State $r' \gets 0$, $\Delta \gets room$
            \Else
                \State $r' \gets (w.G-gap)\cdot w.r / w.G$, $\Delta \gets gap$
            \EndIf
        \EndIf

        \State $loss_w \gets \max(0, w.G-w.N)$

        \If{$selCount+1 > Q_{\max}$ \textbf{or} $selCost+w.C > C_{\max}^{g}$ \textbf{or} $selLoss+loss_w > L_{\max}^{g}$}
            \State \textbf{continue}
        \EndIf
        \If{$batCost+w.C > b.C_{\max}$ \textbf{or} $batLoss+loss_w > b.L_{\max}$}
            \State \textbf{continue}
        \EndIf

        \State Accept $w$: set new regime $r'(w)$ in $S$
        \State $gap \gets gap-\Delta$
        \State Add $w$ to $selected_b$ and global selected set
        \State $selCount \gets selCount+1$, $selCost \gets selCost+w.C$, $selLoss \gets selLoss+loss_w$
        \State $batCost \gets batCost+w.C$, $batLoss \gets batLoss+loss_w$
    \EndFor

    \State Compute achieved production for battery $b$ using adjusted regimes for selected wells
    \If{achieved production not in $[target(1-\tau),\, target(1+\tau)]$}
        \State \Return \textsc{Infeasible}
    \EndIf
\EndFor

\State Set total cost/loss from selected wells
\State Build multi-crew routes with nearest-neighbor + 2-opt over selected wells
\State Store routes, total distance, and mark $S$ as feasible
\State \Return $S$

\end{algorithmic}
\end{algorithm}

\Needspace{4cm}
\begin{algorithm}[H]
\caption{Multi-crew route construction (nearest-neighbor + 2-opt)}
\label{alg:multi_crew_route}
\begin{algorithmic}[1]
\Require Selected well IDs $V$, distance matrix $D$, number of crews $k$
\Ensure Crew routes $R_1,\dots,R_k$ and total traveled distance $dist$

\If{$V = \emptyset$}
    \State \Return $k$ empty routes and $dist \gets 0$
\EndIf

\State Sort $V$ by increasing distance to depot node $0$ using $D[0][v]$
\State Initialize $k$ empty buckets $B_1,\dots,B_k$

\For{$i \gets 1$ to $|V|$}
    \State Assign $V[i]$ to $B_{((i-1) \bmod k)+1}$ 
\EndFor

\State $dist \gets 0$
\For{$c \gets 1$ to $k$}
    \If{$B_c = \emptyset$}
        \State $R_c \gets [0,0]$
        \State \textbf{continue}
    \EndIf

    \State Build an initial route with nearest-neighbor starting at depot $0$
    \State Apply 2-opt local search to improve the route
    \State Ensure route starts and ends at depot: $R_c = [0,\dots,0]$
    \State $dist \gets dist + \sum_{j=1}^{|R_c|-1} D[R_c[j], R_c[j+1]]$
\EndFor

\State \Return $\{R_c\}_{c=1}^{k}$ and $dist$

\end{algorithmic}
\end{algorithm}



\section{Harmony Search}

Harmony Search (HS) was introduced in \cite{geem_harmony_search} as a metaheuristic inspired by musical improvisation, where each decision variable is analogous to a musician's pitch and a candidate solution is a harmony. The original purpose of HS was to provide a flexible global optimization framework for complex engineering design problems, particularly when the search space is nonlinear, nonconvex, and difficult to treat with exact or gradient-based methods.  

The optimization problem proposed in this paper has these same characteristics: (i) coupled battery-level and global constraints, (ii) mixed decision structure (continuous regime values and implicit discrete well selection through regime changes), and (iii) an embedded routing component whose cost depends on the selected wells and therefore introduces additional nonlinearity. For this reason, HS is relevant as a robust exploration--exploitation strategy that can navigate infeasible and feasible regions while progressively improving multi-criteria performance.  

In the implemented solver, each harmony is encoded as a full regime vector $\mathbf{r}'=(r'_1,\dots,r'_{|W|})$, indexed by well id. A well is interpreted as selected when $r'_w \neq r_w$ (current regime), so selection and regime assignment are handled simultaneously. To improve stability and feasibility, candidates are first clamped to $[0,100]$ and then repaired per battery so that achieved production is moved toward the target tolerance interval. This repair operator is a key adaptation for this domain because battery production targets are hard operational requirements.  

Evaluation uses a normalized weighted objective (cost, production loss, route distance, and number of selected wells) plus a penalty term for violations of all constraints (battery targets, battery cost/loss, and global quantity/cost/loss). Normalization avoids scale dominance between heterogeneous terms, and penalties allow controlled exploration near constraint boundaries. The algorithm explicitly tracks both the best overall harmony and the best feasible harmony; at termination, only a feasible harmony is accepted. This design preserves exploration pressure while enforcing operational admissibility in the final recommendation.  

Additional practical adaptations include optional seeding with the greedy solution, which accelerates convergence, and random initialization around current regimes, which keeps early search close to operationally plausible settings. During improvisation, HMCR/PAR/$bw$ regulate memory reuse and local pitch adjustment, while a change probability controls random diversification. Replacing only the worst harmony at each iteration maintains memory quality without losing diversity too quickly.  

From an operational perspective, this HS variant should be interpreted as a constraint-aware improvement heuristic: it does not guarantee global optimality, but it consistently produces feasible, high-quality solutions for instances where exact optimization can be computationally expensive.

\begin{algorithm}[H]
\caption{Harmony Search solver for regime adjustment (core loop)}
\label{alg:harmony_search_solver}
\begin{algorithmic}[1]
\Statex
\State \textbf{Function} $Eval(h)$:
\Statex \hspace{1em} Clamp all regimes to $[0,100]$ and repair battery production toward tolerance window
\Statex \hspace{1em} Extract selected wells ($r'_w \neq r_w$), then compute cost, loss, and multi-crew distance
\Statex \hspace{1em} Compute violations: battery target/cost/loss and global quantity/cost/loss caps
\Statex \hspace{1em} $f(h) \gets$ weighted normalized objective $+\ \lambda\cdot$ normalized violations
\Statex \hspace{1em} Mark feasible iff all violations are zero

\State Initialize harmony memory $HM$ with $HMS$ candidates
\Statex \hspace{1em} (optionally insert one seed from Greedy solution, then fill the rest at random)
\ForAll{$h \in HM$} 
    \State Evaluate $h$ with $Eval(h)$
\EndFor

\State Keep best harmony $h^{best}$ and best feasible harmony $h^{feas}$ (if any)

\For{$t \gets 1$ to $Iter$}
    \State Build new harmony $h^{new}$ well by well:
    \Statex \hspace{1em} with probability $HMCR$, copy regime from a random harmony in $HM$
    \Statex \hspace{1em} and with probability $PAR$, apply pitch adjustment $\pm bw$
    \Statex \hspace{1em} otherwise sample random/new-or-current regime controlled by $p_{chg}$

    \State Evaluate $h^{new}$ with $Eval(h^{new})$
    \State Replace worst harmony in $HM$ if $f(h^{new})$ is better
    \State Update $h^{best}$ and $h^{feas}$
\EndFor

\If{no feasible harmony was found}
    \State \Return \textsc{Infeasible}
\EndIf

\State Decode selected harmony into $S$ (new regimes, selected wells, crew routes, distance, cost, loss)
\State Mark $S$ feasible and \Return $S$

\end{algorithmic}
\end{algorithm}
